def self.from_oauth_token(token)
  return nil unless token && token.token.present?
  access_key, secret_key = token.token.split(':')
  find_by_access_key access_key
end

def in_scopes?(ary)
  return true if ary.blank?
  return true if self[:scopes] == 'all'
  (ary & scopes).present?
end
